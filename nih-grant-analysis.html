df = pd.read_csv('./nih_success_rates/nih_success_data.csv')
df['Percent'] = df['Percent'].str.strip('%')
df['Applications'] = df['Applications'].str.replace(',', '')
df = df.apply(pd.to_numeric)
df['Percent'] *= .01

df['total_hours_spent'] = df['Applications'] * hours_per_application * people_per_application
df['total_hours_awarded'] = df['total_hours_spent'] * df['Percent']
df['total_hours_not_awarded'] = df['total_hours_spent'] * (1 - df['Percent'])

scientist_hours_per_year = 52 * 5 * 8
df['scientist_years'] = df['total_hours_not_awarded'] / scientist_hours_per_year
